```bash
#!/bin/bash
# test_interfaces.sh

interfaces=(
    "http://www.饭太硬.com/tv"
    "https://tvkj.top/DC.txt"
    "https://raw.githubusercontent.com/xyq254245/xyqonlinerule/main/XYQTVBox.json"
)

for url in "${interfaces[@]}"; do
    status=$(curl -s -o /dev/null -w "%{http_code}" --connect-timeout 5 --max-time 10 "$url")
    if [ "$status" = "200" ]; then
        echo "✅ $url"
    else
        echo "❌ $url (状态码: $status)"
    fi
done
```
